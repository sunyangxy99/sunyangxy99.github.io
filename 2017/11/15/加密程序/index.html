import java.io.IOException;
import java.io.File;
import java.io.FileInputStream;
import java.io.FileOutputStream;
import java.io.BufferedInputStream;
import java.io.BufferedOutputStream;
import java.util.Scanner;
class test1
{
public static void main(String[] args)
{
sop(“*欢迎使用加密软件**“);
sop(“*请输入要加密文件的路径**“);
String path=MyInstance.getInstance().next();
sop(“**1.加密 2.解密**“);
int action=MyInstance.getInstance().nextInt();
sop(“请输入密码”);
String password=MyInstance.getInstance().next();
File file=null;
switch(action){
case 1:
file=encryption(path,password);
break;
case 2:
file=deciphering(path,password);
break;
}
if(file!=null)
sop((action==1?”加”:”解”)+”密成功！”+file.getAbsolutePath());
else
sop((action==1?”加”:”解”)+”密失败！”);
}
//打印封装
public static void sop(String str){
System.out.println(str);
}
//加密函数
public static File encryption(String path,String password){
File infile=new File(path);
File outfile=new File(infile.getName());
BufferedInputStream in=null;
BufferedOutputStream out=null;
try{
in=new BufferedInputStream(new FileInputStream(infile));
out=new BufferedOutputStream(new FileOutputStream(outfile));
byte[] pawbytes=password.getBytes();
int len=0;
int i=0;
while((len=in.read())!=-1){
byte b=(byte)len;
out.write(b^pawbytes[i]);
i++;
if(i==pawbytes.length)
i=0;}
return outfile;
}catch(IOException e){
e.printStackTrace();}
catch(NullPointerException e){
e.printStackTrace();
}finally{
try{if(in!=null)
in.close();
if(out!=null)
out.flush();
out.close();}
catch(IOException e){
e.printStackTrace();
}
}
return null;
}
//解密函数
public static File deciphering(String path,String password){
return encryption(path,password);
}
}
//设计一个输入的单例设计模式类
class MyInstance
{
private static MyInstance instance;
private Scanner in=new Scanner(System.in);
private MyInstance(){}
public static MyInstance getInstance(){
if(instance==null)
instance=new MyInstance();
return instance;
}
public String next(){
return in.next();
}
public int nextInt(){
return in.nextInt();
}

}
